# Find all directories that have a given string (i.e. usercube) in the name
find . -type d -iname "*usercube*" 2>/dev/nulls


# Change all filename names that contain a string to the name name but with a new string in that string's place
find . -type f -iname "*usercube*" 2>/dev/null | while read file; do newfile="${file//usercube/identitymanager}"; mv "$file" "$newfile"; echo "Renamed: $file -> $newfile"; done

# command to count how many subdirectories, recursively, in the dir where the command is ran
find . -type d | wc -l

# count how many markdown files in a dir (and all subdirs)
find . -type f -name "*.md" | wc -l
